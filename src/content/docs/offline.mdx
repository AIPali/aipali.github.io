<script is:inline>{`
  const CACHE_NAME = 'aipali-offline-cache';
  let isSyncing = false;

  const UI = {
    log: function(msg) {
      const el = document.getElementById('log-window');
      const time = new Date().toLocaleTimeString('en-GB');
      const lines = el.innerText.split(/[\\r\\n]+/);
      if (lines.length > 50) lines.shift();
      const newline = String.fromCharCode(10);
      el.innerText = lines.join(newline) + newline + '[' + time + '] ' + msg;
      el.scrollTop = el.scrollHeight;
    },
    updateCount: function(count) {
      document.getElementById('cache-count').innerText = count;
    },
    updateStorage: function(usage) {
      document.getElementById('storage-usage').innerText = usage;
    },
    showProgress: function(show) {
      const el = document.getElementById('progress-container');
      if (show) {
        el.classList.remove('hidden');
        void el.offsetWidth;
        el.classList.add('visible');
      } else {
        el.classList.remove('visible');
        setTimeout(function() { el.classList.add('hidden'); }, 300);
      }
    },
    setProgress: function(current, total, text) {
      const percent = total > 0 ? Math.round((current / total) * 100) : 0;
      document.getElementById('progress-bar-fill').style.width = percent + '%';
      document.getElementById('progress-percent').innerText = percent + '%';
      document.getElementById('progress-text').innerText = text;
    },
    disableButtons: function(disabled) {
      document.getElementById('btn-update').disabled = disabled;
      document.getElementById('btn-rebuild').disabled = disabled;
      document.getElementById('btn-clear').disabled = disabled;
    }
  };

  // 严格统计与刷新
  async function refreshStatus() {
    try {
      if ('caches' in window) {
        const hasCache = await caches.has(CACHE_NAME);
        if (hasCache) {
          const cache = await caches.open(CACHE_NAME);
          const keys = await cache.keys();
          // 只统计真正的 HTML 页面，过滤掉随之缓存的 CSS/JS/IMG，解决数字膨胀假象
          const htmlCount = keys.filter(function(req) {
            const url = new URL(req.url);
            return url.pathname.endsWith('/') || url.pathname.endsWith('.html');
          }).length;
          UI.updateCount(htmlCount);
        } else {
          UI.updateCount(0);
        }
      } else {
        UI.updateCount('N/A');
        UI.log('❌ 浏览器不支持缓存引擎。');
      }

      if (navigator.storage && navigator.storage.estimate) {
        const estimate = await navigator.storage.estimate();
        const usedMB = (estimate.usage / (1024 * 1024)).toFixed(2);
        UI.updateStorage(usedMB);
      } else {
        UI.updateStorage('--');
      }
    } catch (e) {
      UI.log('状态获取异常: ' + e.message);
    }
  }

  // 强制轮询刷新：iOS 测算容量有延迟，执行任务后连续刷新 3 次确保 UI 更新
  function triggerRefresh() {
    refreshStatus();
    setTimeout(refreshStatus, 1500);
    setTimeout(refreshStatus, 3000);
  }

  window.clearAllCache = async function() {
    if (isSyncing) return;
    const msg = '【彻底清除确认】\\n\\n此操作将完全抹除设备上所有已下载的经文数据。\\n等同于应用“重置”。断网时您将无法阅读。\\n\\n确定要继续吗？';
    if (!confirm(msg)) return;
    
    UI.log('开始销毁本地底层数据区块...');
    try {
      await caches.delete(CACHE_NAME);
      UI.log('✅ 核心数据已清空。应用已恢复至纯净初始状态。');
      triggerRefresh();
    } catch (e) {
      UI.log('❌ 清理失败: ' + e.message);
    }
  };

  window.startSync = async function(isRebuild) {
    if (isSyncing) return;
    
    let msg = isRebuild 
      ? '【全量重建确认】\\n\\n此操作将首先清空所有旧缓存（包含旧版样式及废弃经文），然后重新下载全站数据。\\n\\n若发现页面错乱或体积异常膨胀，请使用此功能。确定开始吗？'
      : '【智能增量确认】\\n\\n此操作仅下载新增或被修改过的经文，极度节省流量。\\n\\n确定开始吗？';
      
    if (!confirm(msg)) return;

    isSyncing = true;
    UI.disableButtons(true);
    UI.showProgress(true);
    
    UI.log('');
    UI.log(isRebuild ? '>>> 启动全量重建协议 (Rebuild)' : '>>> 启动智能增量协议 (Update)');

    try {
      if (!navigator.onLine) {
        throw new Error('设备处于离线状态，无法连接到云端。');
      }

      UI.log('正在解析云端结构目录 (sitemap-0.xml)...');
      const sitemapRes = await fetch('/sitemap-0.xml', { cache: 'no-store' });
      if (!sitemapRes.ok) throw new Error('Sitemap 响应异常 (HTTP ' + sitemapRes.status + ')');
      
      const sitemapText = await sitemapRes.text();
      const parser = new DOMParser();
      const xml = parser.parseFromString(sitemapText, "text/xml");
      
      const urls = Array.from(xml.querySelectorAll("loc"))
        .map(function(node) {
          try { return new URL(node.textContent).pathname; } 
          catch(e) { return node.textContent; }
        })
        .filter(function(url) {
          // 仅过滤掉 /tags/ 和 /info/，保留根目录 / 和 /offline/
          return !url.includes('/tags/');
        });

      if (urls.length === 0) throw new Error('未发现有效的经文下载节点。');
      UI.log('成功锁定 ' + urls.length + ' 个核心数据节点。');

      const cache = await caches.open(CACHE_NAME);
      let existingKeys = [];
      
      if (!isRebuild) {
        const reqs = await cache.keys();
        existingKeys = reqs.map(function(req) { return new URL(req.url).pathname; });
        UI.log('比对本地 ' + existingKeys.length + ' 个数据块...');
        
        // 垃圾回收：如果开启增量，找出本地有但线上 sitemap 已经没有的冗余页面，删掉！防止膨胀。
        let gcCount = 0;
        for (const req of reqs) {
          const reqPath = new URL(req.url).pathname;
          if (reqPath.endsWith('/') && !urls.includes(reqPath)) {
            await cache.delete(req);
            gcCount++;
          }
        }
        if (gcCount > 0) UI.log('清理了 ' + gcCount + ' 个已废弃的旧版本区块。');
      } else {
        UI.log('正在强制格式化旧的缓存区块...');
        await caches.delete(CACHE_NAME);
        await caches.open(CACHE_NAME); // 重新打开干净的缓存池
      }

      let downloaded = 0;
      let skipped = 0;
      let errors = 0;
      let currentIndex = 0;
      const MAX_CONCURRENT = 5;

      const downloadWorker = async function() {
        while (currentIndex < urls.length) {
          const index = currentIndex++;
          const url = urls[index];
          const displayUrl = url === '/' ? '首页 (Index)' : url.split('/').filter(Boolean).slice(-2).join('/'); 

          if (!isRebuild && existingKeys.includes(url)) {
            skipped++;
          } else {
            try {
              UI.setProgress(currentIndex, urls.length, '正在获取: ' + displayUrl);
              const res = await fetch(url, { cache: 'no-cache' });
              
              if (res.ok) {
                // 防止并发写入报错，重新获取 cache 实例
                const curCache = await caches.open(CACHE_NAME);
                await curCache.put(url, res);
                downloaded++;
              } else {
                UI.log('⚠️ 节点失效 [' + res.status + ']: ' + displayUrl);
                errors++;
              }
            } catch (err) {
              UI.log('⚠️ 网络阻断: ' + displayUrl);
              errors++;
            }
          }
          
          if (currentIndex % 10 === 0 || currentIndex >= urls.length - MAX_CONCURRENT) {
            UI.setProgress(currentIndex, urls.length, '同步中... (' + currentIndex + '/' + urls.length + ')');
          }
        }
      };

      UI.log('启动并发下载矩阵 (Threads: 5)...');
      const workers = Array(Math.min(MAX_CONCURRENT, urls.length)).fill(null).map(function() { return downloadWorker(); });
      await Promise.all(workers);

      UI.setProgress(urls.length, urls.length, '数据同步已完成');
      UI.log('✅ 任务结束 | 新增/更新: ' + downloaded + ' 篇 | 跳过: ' + skipped + ' 篇 | 失败: ' + errors + ' 篇');

    } catch (e) {
      if (e.message.includes('Failed to fetch') || e.message.includes('NetworkError')) {
        UI.log('❌ 同步中止: 设备网络无法连接。');
      } else {
        UI.log('❌ 严重错误，协议中止: ' + e.message);
      }
    } finally {
      isSyncing = false;
      UI.disableButtons(false);
      setTimeout(function() { UI.showProgress(false); }, 3000);
      triggerRefresh();
    }
  };

  document.addEventListener('DOMContentLoaded', function() {
    refreshStatus();
    window.addEventListener('online', function() { UI.log('📶 检测到网络连接已恢复。'); });
    window.addEventListener('offline', function() { UI.log('📴 检测到网络连接已断开，进入离线守护模式。'); });
  });
`}</script>
