<script is:inline>{`
  const CACHE_NAME = 'aipali-offline-cache';
  let isSyncing = false;

  const UI = {
    log: function(msg) {
      const el = document.getElementById('log-window');
      const time = new Date().toLocaleTimeString('en-GB');
      const lines = el.innerText.split(/[\\r\\n]+/);
      if (lines.length > 60) lines.shift();
      const newline = String.fromCharCode(10);
      el.innerText = lines.join(newline) + newline + '[' + time + '] ' + msg;
      el.scrollTop = el.scrollHeight;
    },
    updateCount: function(count) {
      document.getElementById('cache-count').innerText = count;
    },
    updateStorage: function(usage) {
      document.getElementById('storage-usage').innerText = usage === '--' ? '-- MB' : usage + ' MB';
    },
    showProgress: function(show) {
      const el = document.getElementById('progress-container');
      if (show) {
        el.classList.remove('hidden');
        void el.offsetWidth;
        el.classList.add('visible');
      } else {
        el.classList.remove('visible');
        setTimeout(function() { el.classList.add('hidden'); }, 300);
      }
    },
    setProgress: function(current, total, text) {
      const percent = total > 0 ? Math.round((current / total) * 100) : 0;
      document.getElementById('progress-bar-fill').style.width = percent + '%';
      document.getElementById('progress-percent').innerText = percent + '%';
      if (text) document.getElementById('progress-text').innerText = text;
    },
    disableButtons: function(disabled) {
      document.getElementById('btn-update').disabled = disabled;
      document.getElementById('btn-rebuild').disabled = disabled;
      document.getElementById('btn-clear').disabled = disabled;
    }
  };

  async function refreshStatus() {
    try {
      if ('caches' in window) {
        const cacheNames = await caches.keys();
        const pathSet = new Set(); // 使用 Set 严格去重
        
        for (let name of cacheNames) {
          const cache = await caches.open(name);
          const keys = await cache.keys();
          
          for (let req of keys) {
            try {
              const url = new URL(req.url);
              const path = url.pathname;
              // 严格筛选：结尾是 / 或 .html，且排除 astro 生成的资产文件
              if ((path.endsWith('/') || path.endsWith('.html')) && 
                  !path.includes('/_astro/') && 
                  !path.includes('/assets/')) {
                pathSet.add(path);
              }
            } catch(e) {}
          }
        }
        UI.updateCount(pathSet.size);
      } else {
        UI.updateCount('N/A');
        UI.log('系统警告: 当前运行环境不支持 Cache API。');
      }

      if (navigator.storage && navigator.storage.estimate) {
        const estimate = await navigator.storage.estimate();
        const usedMB = (estimate.usage / (1024 * 1024)).toFixed(2);
        UI.updateStorage(usedMB);
      } else {
        UI.updateStorage('--');
      }
    } catch (e) {
      UI.log('状态读取异常: ' + e.message);
    }
  }

  function refreshStatusWithRetry() {
    refreshStatus();
    setTimeout(refreshStatus, 1000);
    setTimeout(refreshStatus, 3000);
  }

  window.confirmAndClear = async function() {
    if (isSyncing) return;
    const msg = "【清除本地数据】\\n此操作将彻底销毁本地所有的法宝节点、应用缓存以及服务组件。清除后，在无网络连接的状态下将无法继续阅读。\\n\\n确认执行最高级别的数据抹除协议？";
    if (!confirm(msg)) return;
    
    UI.log('正在执行全盘数据抹除协议...');
    UI.disableButtons(true);
    try {
      // 1. 删除所有 Cache Storage (包括各类资源)
      const cacheNames = await caches.keys();
      for (const name of cacheNames) {
        await caches.delete(name);
      }
      
      // 2. 清理 Workbox 底层的 IndexedDB 缓存过期机制数据
      if (window.indexedDB && indexedDB.databases) {
        try {
          const dbs = await indexedDB.databases();
          for (const db of dbs) {
            indexedDB.deleteDatabase(db.name);
          }
        } catch(err) {}
      }
      
      // 3. 卸载所有的 Service Worker 保证环境完全纯洁
      if ('serviceWorker' in navigator) {
        const registrations = await navigator.serviceWorker.getRegistrations();
        for (const reg of registrations) {
          await reg.unregister();
        }
      }

      UI.log('本地所有法宝数据及关联缓存已彻底抹除！');
      UI.log('【⚠️ 核心提示】由于浏览器底层存储状态的限制，当前界面的统计数值不会立刻刷新，且部分系统缓存需彻底关闭后释放。请务必 [彻底划掉/退出当前应用] 并重新进入，以查看准确的归零容量。');
      
    } catch (e) {
      UI.log('清除操作执行受阻: ' + e.message);
    } finally {
      UI.disableButtons(false);
      // 清除后不再重试读取，直接强行设定为 --，避免滞后的读数迷惑用户
      UI.updateCount('--');
      UI.updateStorage('--');
    }
  };

  window.confirmAndStartSync = async function(isRebuild) {
    if (isSyncing) return;
    if (!navigator.onLine) {
      alert("【网络未连接】\\n终端当前处于离线状态，无法连接至云端服务器进行数据比对。请恢复网络连接后再试。");
      return;
    }

    const msg = isRebuild
      ? "【全量重新构建】\\n系统将格式化现有数据，并从云端完整拉取所有法宝节点。此过程耗时较长，请确保设备电量充足且网络环境稳定。\\n\\n确认执行全量重建？"
      : "【增量同步数据】\\n系统将比对本地与云端目录拓扑，仅下载新增或变动的法宝节点。\\n\\n确认执行增量同步？";
    
    if (!confirm(msg)) return;
    await performSync(isRebuild);
  };

  async function performSync(isRebuild) {
    isSyncing = true;
    UI.disableButtons(true);
    UI.showProgress(true);
    
    UI.log('');
    UI.log(isRebuild ? '=== 启动全量重建协议 ===' : '=== 启动增量同步协议 ===');

    try {
      if (!navigator.onLine) {
        throw new Error('终端网络意外断开，协议执行终止。');
      }

      UI.log('读取云端拓扑结构 (sitemap-0.xml)...');
      const sitemapRes = await fetch('/sitemap-0.xml', { cache: 'no-store' });
      if (!sitemapRes.ok) throw new Error('云端通信受阻 (HTTP ' + sitemapRes.status + ')');
      
      const sitemapText = await sitemapRes.text();
      const parser = new DOMParser();
      const xml = parser.parseFromString(sitemapText, "text/xml");
      
      const urls = Array.from(xml.querySelectorAll("loc"))
        .map(function(node) {
          try { return new URL(node.textContent).pathname; } 
          catch(e) { return node.textContent; }
        })
        .filter(function(url) {
          return !url.includes('/tags/');
        });

      if (urls.length === 0) throw new Error('未获取到合法的法宝数据节点。');
      UI.log('校验通过，共有 ' + urls.length + ' 个合法同步节点。');

      const cache = await caches.open(CACHE_NAME);
      let existingKeys =[];
      
      if (!isRebuild) {
        const reqs = await cache.keys();
        existingKeys = reqs.map(function(req) { return new URL(req.url).pathname; });
        UI.log('本地探明已缓存节点，开始差异比对...');
      } else {
        UI.log('执行底层格式化，清空旧有数据区块...');
        const cacheNames = await caches.keys();
        for (const name of cacheNames) {
          await caches.delete(name);
        }
      }

      let downloaded = 0;
      let skipped = 0;
      let errors = 0;
      let currentIndex = 0;
      const MAX_CONCURRENT = 5;

      const downloadWorker = async function() {
        while (currentIndex < urls.length) {
          const index = currentIndex++;
          const url = urls[index];
          
          // 格式化输出文件路径：.../前一层/当前层/
          const segments = url.split('/').filter(Boolean);
          const displayUrl = segments.length > 1 ? '.../' + segments.slice(-2).join('/') + '/' : url;

          if (!isRebuild && existingKeys.includes(url)) {
            skipped++;
          } else {
            try {
              UI.setProgress(currentIndex, urls.length, '拉取节点: ' + displayUrl);
              const res = await fetch(url, { cache: 'no-cache' });
              
              if (res.ok) {
                const currentCache = await caches.open(CACHE_NAME);
                await currentCache.put(url, res);
                downloaded++;
              } else {
                UI.log('节点响应失效 [' + res.status + ']: ' + url);
                errors++;
              }
            } catch (err) {
              UI.log('节点传输阻断: ' + url);
              errors++;
            }
          }
          
          // 仅在关键节点或即将结束时刷新全局进度文字（避免闪烁）
          if (currentIndex % 10 === 0 || currentIndex >= urls.length - MAX_CONCURRENT) {
            UI.setProgress(currentIndex, urls.length); 
          }
        }
      };

      UI.log('开启多路复用传输 (线程数: 5)...');
      const workers = Array(Math.min(MAX_CONCURRENT, urls.length)).fill(null).map(function() { return downloadWorker(); });
      await Promise.all(workers);

      UI.setProgress(urls.length, urls.length, '数据同步通道已关闭');
      UI.log('协议执行完毕 | 新增/更新: ' + downloaded + ' | 跳过: ' + skipped + ' | 异常: ' + errors);

    } catch (e) {
      UI.log('错误: ' + e.message);
    } finally {
      isSyncing = false;
      UI.disableButtons(false);
      setTimeout(function() { UI.showProgress(false); }, 3000);
      refreshStatusWithRetry();
    }
  }

  document.addEventListener('DOMContentLoaded', function() {
    refreshStatus();
    window.addEventListener('online', function() { UI.log('系统侦测到网络连接，同步模块就绪。'); });
    window.addEventListener('offline', function() { UI.log('系统侦测到网络中断，已平滑切入全离线单机模式。'); });
  });
`}</script>
