<script is:inline>
  const CACHE_NAME = 'aipali-html-cache';
  let isSyncing = false;

  const UI = {
    log: (msg) => {
      const el = document.getElementById('log-window');
      const time = new Date().toLocaleTimeString('en-GB');
      // 自动追加日志，保留最新的 50 行防止内存溢出
      const lines = el.innerText.split('\n');
      if (lines.length > 50) lines.shift();
      el.innerText = lines.join('\n') + `\n[${time}] ${msg}`;
      el.scrollTop = el.scrollHeight; // 滚动到底部
    },
    updateCount: (count) => document.getElementById('cache-count').innerText = count,
    updateStorage: (usage) => document.getElementById('storage-usage').innerText = usage,
    showProgress: (show) => {
      const el = document.getElementById('progress-container');
      if (show) {
        el.classList.remove('hidden');
        // 强制重绘，触发动画
        void el.offsetWidth;
        el.classList.add('visible');
      } else {
        el.classList.remove('visible');
        setTimeout(() => el.classList.add('hidden'), 300); // 等待淡出动画结束
      }
    },
    setProgress: (current, total, text) => {
      const percent = total > 0 ? Math.round((current / total) * 100) : 0;
      document.getElementById('progress-bar-fill').style.width = `${percent}%`;
      document.getElementById('progress-percent').innerText = `${percent}%`;
      document.getElementById('progress-text').innerText = text;
    },
    disableButtons: (disabled) => {
      document.getElementById('btn-update').disabled = disabled;
      document.getElementById('btn-rebuild').disabled = disabled;
      document.getElementById('btn-clear').disabled = disabled;
    }
  };

  // 初始化：获取缓存详情
  async function refreshStatus() {
    try {
      // 1. 统计经文数量
      if ('caches' in window) {
        const hasCache = await caches.has(CACHE_NAME);
        if (hasCache) {
          const cache = await caches.open(CACHE_NAME);
          const keys = await cache.keys();
          UI.updateCount(keys.length);
        } else {
          UI.updateCount(0);
        }
      } else {
        UI.updateCount('N/A');
        UI.log('❌ 当前浏览器不支持 Cache API。');
      }

      // 2. 统计硬盘空间
      if (navigator.storage && navigator.storage.estimate) {
        const estimate = await navigator.storage.estimate();
        const usedMB = (estimate.usage / (1024 * 1024)).toFixed(2);
        UI.updateStorage(usedMB);
      } else {
        UI.updateStorage('--');
      }
    } catch (e) {
      UI.log('⚠️ 状态获取异常: ' + e.message);
    }
  }

  // 清理缓存
  window.clearAllCache = async () => {
    if (isSyncing) return;
    if (!confirm('警告：此操作将清空设备上所有已下载的经文数据。\n断网时您将无法阅读未缓存的内容。\n\n确定要继续吗？')) return;
    
    UI.log('开始销毁本地数据区块...');
    try {
      await caches.delete(CACHE_NAME);
      UI.log('✅ 所有离线数据已成功清空。');
      refreshStatus();
    } catch (e) {
      UI.log('❌ 清理失败: ' + e.message);
    }
  };

  // 核心：智能/全量下载逻辑
  window.startSync = async (isRebuild) => {
    if (isSyncing) return;
    isSyncing = true;
    UI.disableButtons(true);
    UI.showProgress(true);
    
    UI.log('');
    UI.log(isRebuild ? '>>> 启动全量重建协议 (Rebuild)' : '>>> 启动智能增量协议 (Update)');

    try {
      // 1. 检查网络
      if (!navigator.onLine) {
        throw new Error('设备处于离线状态，无法连接到云端服务器。');
      }

      // 2. 获取站点地图 (Sitemap)
      UI.log('正在解析云端结构目录 (sitemap-0.xml)...');
      const sitemapRes = await fetch('/sitemap-0.xml', { cache: 'no-store' });
      if (!sitemapRes.ok) throw new Error(`Sitemap 响应异常 (HTTP ${sitemapRes.status})`);
      
      const sitemapText = await sitemapRes.text();
      const parser = new DOMParser();
      const xml = parser.parseFromString(sitemapText, "text/xml");
      
      // 提取链接，严苛过滤：排除 tags、info，仅保留真实的经文和必需页面
      const urls = Array.from(xml.querySelectorAll("loc"))
        .map(node => {
          try {
            return new URL(node.textContent).pathname; // 仅截取路径部分，避免跨域冲突
          } catch(e) {
            return node.textContent;
          }
        })
        .filter(url => 
          !url.includes('/tags/') && 
          !url.includes('/info/') && 
          !url.includes('/offline/') && 
          url !== '/' && // 排除首页
          url.length > 2
        );

      if (urls.length === 0) throw new Error('未发现有效的经文下载节点。');
      UI.log(`成功锁定 ${urls.length} 个核心经文节点。`);

      const cache = await caches.open(CACHE_NAME);
      let existingKeys = [];
      
      if (!isRebuild) {
        // 增量模式：比对现有缓存
        const reqs = await cache.keys();
        // 将 Request 对象的 URL 转换为 pathname 用于精准比对
        existingKeys = reqs.map(req => new URL(req.url).pathname);
        UI.log(`本地已发现 ${existingKeys.length} 个缓存区块，开始校验差异...`);
      } else {
        // 全量重建：销毁旧区块
        UI.log('正在格式化旧的缓存区块...');
        const reqs = await cache.keys();
        for (const req of reqs) {
          await cache.delete(req);
        }
      }

      // 3. 开始多线程并发下载引擎 (控制并发数为 5，防止压死浏览器/服务器)
      let downloaded = 0;
      let skipped = 0;
      let errors = 0;
      let currentIndex = 0;
      const MAX_CONCURRENT = 5;

      // 并发下载工头函数
      const downloadWorker = async () => {
        while (currentIndex < urls.length) {
          const index = currentIndex++;
          const url = urls[index];
          
          // 为了 UI 显示美观，只截取路径最后一段
          const displayUrl = url.split('/').filter(Boolean).slice(-2).join('/'); 

          if (!isRebuild && existingKeys.includes(url)) {
            skipped++;
          } else {
            try {
              UI.setProgress(currentIndex, urls.length, `正在获取: .../${displayUrl}`);
              
              // 强制刷新网络请求，保证拿到云端最新校对版本
              const res = await fetch(url, { cache: 'no-cache' });
              
              if (res.ok) {
                await cache.put(url, res);
                downloaded++;
              } else {
                UI.log(`⚠️ 节点失效 [${res.status}]: ${url}`);
                errors++;
              }
            } catch (err) {
              UI.log(`⚠️ 网络阻断: ${url}`);
              errors++;
            }
          }
          
          // UI 节流渲染：每 10 次或者快结束时更新一次界面，防止主线程卡顿
          if (currentIndex % 10 === 0 || currentIndex >= urls.length - MAX_CONCURRENT) {
            UI.setProgress(currentIndex, urls.length, `进度: ${currentIndex}/${urls.length}`);
            refreshStatus();
          }
        }
      };

      UI.log('启动并发下载矩阵 (Threads: 5)...');
      // 启动 5 个并发工人
      const workers = Array(Math.min(MAX_CONCURRENT, urls.length)).fill(null).map(() => downloadWorker());
      await Promise.all(workers);

      UI.setProgress(urls.length, urls.length, '数据同步已完成');
      UI.log(`✅ 任务结束 | 新增/更新: ${downloaded} 篇 | 跳过: ${skipped} 篇 | 失败: ${errors} 篇`);

    } catch (e) {
      UI.log('❌ 严重错误，协议中止: ' + e.message);
    } finally {
      isSyncing = false;
      UI.disableButtons(false);
      setTimeout(() => UI.showProgress(false), 3000);
      refreshStatus();
    }
  };

  // 页面加载完成执行
  document.addEventListener('DOMContentLoaded', () => {
    refreshStatus();
    // 监听网络状态变化
    window.addEventListener('online',  () => UI.log('📶 检测到网络连接已恢复。'));
    window.addEventListener('offline', () => UI.log('📴 检测到网络连接已断开，进入离线守护模式。'));
  });
</script>
